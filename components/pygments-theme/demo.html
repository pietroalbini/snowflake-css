import os
import sys

def count_files(path):
    # Return the number of files in a directory
    return len(os.listdir(path))

if __name__ == "__main__":
    if len(sys.argv) == 1:
        total = count_files(".")
    else:
        total = sum(count_files(d) for d in sys.argv[1:])
    print("Total: %s" % total)
